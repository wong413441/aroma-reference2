// 宣告全域變數 let AROMA_DATA = { OIL_DATABASE: {}, BLENDS_LIBRARY: [] }; // 頁面載入時自動讀取 data.json window.onload = async function() { try { const response = await fetch('data.json'); AROMA_DATA = await response.json(); // 載入完成後，自動渲染配方列表 renderB(AROMA_DATA.BLENDS_LIBRARY); } catch (error) { console.error("無法載入資料庫，請確認 data.json 是否已上傳至 GitHub。", error); } }; // 配方渲染函數（加入滑鼠懸停顯示機制的功能） function renderB(rows) { const grid = document.getElementById('blend-grid'); grid.innerHTML = ''; rows.forEach(r => { const card = document.createElement('div'); card.className = 'blend-card'; // 生成標籤 const routeHtml = r.routes.map(rt => `${rt}`).join(''); // 生成精油列表，並查詢其治療機制 const oilsHtml = r.oils.map(o => { // 從名稱中提取純中文名來比對資料庫 (例如 "熏陸香 Pistacia lentiscus" -> "熏陸香") const oilKey = o.n.split(' ')[0]; const info = AROMA_DATA.OIL_DATABASE[oilKey] || { function: "機制詳見化合物列表" }; return `
${o.n} ${o.p}
`; }).join(''); // 組合卡片 HTML card.innerHTML = `
${r.name}
${r.sys}
${oilsHtml}
${routeHtml}
${r.note ? `
${r.note}
` : ''}`; grid.appendChild(card); }); document.getElementById('b-count').textContent = rows.length + ' 個配方'; }
